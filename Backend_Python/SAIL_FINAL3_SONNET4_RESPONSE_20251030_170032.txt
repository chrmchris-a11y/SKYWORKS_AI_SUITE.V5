from sail.validators.sail_validator import SAILValidator
from sail.models.sail_models import SAILInputs20, SAILInputs25, ARCValue


class TestSAILValidation:
    """Test SAIL input validation functionality."""
    
    def setup_method(self):
        """Setup test fixtures."""
        self.validator_20 = SAILValidator(sora_version="2.0")
        self.validator_25 = SAILValidator(sora_version="2.5")
    
    def test_valid_inputs_20(self):
        """Test validation of valid inputs for SORA 2.0."""
        valid_inputs = [
            SAILInputs20(grc=1, arc=ARCValue.A),
            SAILInputs20(grc=4, arc=ARCValue.B),
            SAILInputs20(grc=8, arc=ARCValue.D),
            SAILInputs20(grc=5, arc=ARCValue.C)
        ]
        
        for inputs in valid_inputs:
            # Should not raise any exceptions
            self.validator_20.validate_inputs(inputs)
            assert self.validator_20.is_valid_grc(inputs.grc)
            assert self.validator_20.is_valid_arc(inputs.arc)
    
    def test_valid_inputs_25(self):
        """Test validation of valid inputs for SORA 2.5."""
        valid_inputs = [
            SAILInputs25(grc=1, arc=ARCValue.A),
            SAILInputs25(grc=4, arc=ARCValue.B),
            SAILInputs25(grc=8, arc=ARCValue.D),
            SAILInputs25(grc=5, arc=ARCValue.C)

        for inputs in valid_inputs:
            # Should not raise any exceptions

            self.validator_25.validate_inputs(invalid_inputs_25)

        assert not self.validator_25.is_valid_grc(0)

        invalid_inputs_20 = SAILInputs20(grc=9, arc=ARCValue.A)
        invalid_inputs_25 = SAILInputs25(grc=9, arc=ARCValue.A)
        
        with pytest.raises(ValueError, match="GRC must be between 1 and 8"):
            self.validator_20.validate_inputs(invalid_inputs_20)

"""
SAIL (Specific Assurance and Integrity Level) Calculator Package

This package provides implementations for calculating SAIL values according to
SORA (Specific Operations Risk Assessment) 2.0 and 2.5 standards.

The SAIL value is determined based on:
- GRC (Ground Risk Class): Integer value from 1 to 8
- ARC (Air Risk Class): Enumerated value (A, B, C, or D)

Key Components:
- SAILCalculator20/25: Main calculation engines
- SAILInputs20/25: Input validation models
- SAILResult: Calculation results with OSO requirements
- OSOMapper: Maps SAIL levels to OSO requirements
- SAILValidator: Input validation utilities
"""

# Convenience functions for quick access
def calculate_sail_20(grc: int, arc: ARCValue) -> SAILResult:
    """
    Quick calculation function for SORA 2.0.
    
    Args:
        grc: Ground Risk Class (1-8)
        arc: Air Risk Class (A, B, C, or D)
        
    Returns:
        SAILResult with calculated SAIL value and OSO requirements
        
    Example:
        >>> from sail import calculate_sail_20, ARCValue
        >>> result = calculate_sail_20(4, ARCValue.B)
        >>> print(f"SAIL: {result.sail_value}")
    """
    calculator = SAILCalculator20()
    inputs = SAILInputs20(grc=grc, arc=arc)
    return calculator.calculate_sail(inputs)
